START
    DECLARE int num1, num2, num3
    PROMPT user to enter three integers
    READ num1, num2, num3

    DECLARE integer pointers ptr1, ptr2, ptr3
    ALLOCATE memory dynamically for ptr1, ptr2, ptr3 using 'new' operator

    STORE values in allocated memory
    *ptr1 = num1
    *ptr2 = num2
    *ptr3 = num3

    PRINT "Values entered: ", num1, num2, num3
    PRINT "Pointer values: ", *ptr1, *ptr2, *ptr3
    PRINT "Memory addresses: ", ptr1, ptr2, ptr3

    DEALLOCATE memory using 'delete' operator
    PRINT "Memory has been freed."
END
